1. **Add template API functions to `frontend/src/lib/api.ts`:**

   Add a new section after the Generation API section (added in Plan 02). If Plan 02 has not added its section yet, add after the Admin API section:

   ```typescript
   // =============================================================================
   // Document Templates API (Admin)
   // =============================================================================

   export interface DocumentTemplate {
     id: number;
     name: string;
     template_type: string;
     system_prompt: string;
     user_prompt_template: string;
     is_active: boolean;
     created_at: string;
     updated_at: string;
   }

   export const templatesApi = {
     list: async (): Promise<DocumentTemplate[]> => {
       return api.get<DocumentTemplate[]>("/admin/templates");
     },
     get: async (id: number): Promise<DocumentTemplate> => {
       return api.get(`/admin/templates/${id}`);
     },
     update: async (id: number, data: {
       name?: string;
       system_prompt?: string;
       user_prompt_template?: string;
       is_active?: boolean;
     }): Promise<DocumentTemplate> => {
       return api.put(`/admin/templates/${id}`, data);
     },
   };
   ```

      ```typescript
      const handleTemplateSave = async () => {
        if (!editingTemplate) return;
        setTemplateSaving(true);
        setTemplateError(null);
        try {
          const updated = await templatesApi.update(editingTemplate.id, {
            name: editingTemplate.name,
            system_prompt: editingTemplate.system_prompt,
            user_prompt_template: editingTemplate.user_prompt_template,
            is_active: editingTemplate.is_active,
          });
          setTemplates(prev => prev.map(t => t.id === updated.id ? updated : t));
          setEditingTemplate(null);
        } catch (err) {
          setTemplateError(err instanceof Error ? err.message : "Failed to save template");
        } finally {
          setTemplateSaving(false);
        }
      };
      ```
