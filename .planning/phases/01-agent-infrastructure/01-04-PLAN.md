```typescript
  | { type: "error"; message: string }
  | { type: "usage"; prompt_tokens: number; completion_tokens: number; model: string }
  | { type: "done" };
```

    2. Create `frontend/src/lib/agent-types.ts` with shared TypeScript types:
       ```typescript
       export type AgentEvent =
         | { type: "start"; agent_name: string }
         | { type: "text_delta"; content: string }
         | { type: "tool_start"; tool_name: string; tool_call_id: string }
         | { type: "tool_result"; tool_name: string; tool_call_id: string; result: string; success: boolean }
         | { type: "error"; message: string }
         | { type: "usage"; prompt_tokens: number; completion_tokens: number; model: string }
         | { type: "done" };

       export interface ToolCallEvent {
         id: string;
         name: string;
         status: "executing" | "completed" | "failed";
         result?: string;
       }

       export interface AgentStreamState {
         status: "idle" | "streaming" | "done" | "error";
         agentName: string;
         text: string;
         toolCalls: ToolCallEvent[];
         error: string | null;
         usage: { prompt_tokens: number; completion_tokens: number; model: string } | null;
       }
       ```

    3. Create `frontend/src/hooks/use-agent-stream.ts`:
       - Export `useAgentStream(options?: { onToolComplete?: () => void })` hook
       - Internal state: `AgentStreamState` via useState, initial status "idle"
       - AbortController ref for cancellation (prevents stale closures)
       - `run(agentSlug: string, message: string, userEmail: string, context?: Record<string, unknown>)` async function:
         a. Abort any in-flight request via ref
         b. Create new AbortController, store in ref
         c. Reset state to streaming, clear text/toolCalls/error
         d. `fetch("/api/proxy/agents/run", { method: "POST", headers: {"Content-Type": "application/json", "X-User-Email": userEmail}, body: JSON.stringify({agent_slug: agentSlug, message, context}), signal: controller.signal })`
         e. Check response.ok, throw on error
         f. Get reader from response.body.getReader()
         g. Buffer-based SSE parsing loop:
            - Read chunks via reader.read()
            - Decode with TextDecoder({stream: true})
            - Accumulate in buffer string
            - Split by "\n", keep last incomplete line in buffer
            - For each complete line starting with "data: ", parse JSON
            - Call handleEvent for each parsed event
         h. On completion, set status to "done"
         i. On error (not AbortError), set status to "error" with message
       - `handleEvent(event: AgentEvent)` function:
         - "start": set agentName
         - "text_delta": append content to text
         - "tool_start": add new ToolCallEvent with status "executing"
         - "tool_result": update matching ToolCallEvent status to "completed"/"failed", set result. Call options.onToolComplete?.() to trigger page data refresh.
         - "error": set error message, set status to "error"
         - "usage": store usage data
         - "done": set status to "done"
       - `cancel()` function: abort controller, reset status to "idle"
       - `reset()` function: reset entire state to initial
       - Return: `{ ...state, run, cancel, reset }`
       - Cleanup: useEffect return that calls cancel on unmount

       IMPORTANT: The hook accepts userEmail as a run() parameter (not via useSession). This keeps the hook decoupled from NextAuth. The parent component (AgentResponsePanel) provides userEmail from its props.
